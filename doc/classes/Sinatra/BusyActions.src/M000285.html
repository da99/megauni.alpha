        def busy_action_for(  raw_class_name, perm_level, action_name, path = nil, &custom_action )

            # === Set default values.
            class_name = raw_class_name.to_sym
            path ||= /\/#{class_name.to_s.underscore}\/#{action_name.to_s.underscore}([\/\d]+)?$/

            # === Validate action name.
            busy_action_validate!( action_name )            
            
            # === Make sure class name and action name have not 
            # === been used before.
            self.busy_action_cache[class_name] ||= []
            if self.busy_action_cache[class_name].include?(action_name)
                raise " #{action_name.inspect}, already used for,  #{class_name.inspect}" 
            end
            self.busy_action_cache[class_name] << action_name
            
            # === Create the actions.
            case action_name
                when :INDEX, :NEW, :EDIT
                    method_type = :GET
                when :UPDATE, :TRASH, :UNTRASH
                    method_type = :PUT
                when :CREATE
                    method_type = :POST
                else
                    raise "Programmer error: Forgot to add specify method name for busy action: #{action_name}"
            end
            
            # === Create Sinatra actions like:  
            #  get( "/newspaper" ) { 
            #      do_busy_action 
            #  } 
            eval( %~
                #{method_type.to_s.underscore}_it_for( #{perm_level.inspect}, #{path.inspect} ) do
                    describe_action #{class_name}, "#{action_name}"
                    def self.do_busy_action
                        send( 'busy_#{action_name.to_s.underscore}' )
                    end
                    
                    #{ 
                        custom_action ?
                            strigify_proc(custom_action): 
                            'do_busy_action' 
                      }
                    
                end
            ~)
          
        end
