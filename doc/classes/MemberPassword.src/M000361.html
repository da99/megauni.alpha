    def self.included(target)
    
        target.extend ClassMethods
        
        target.hook_for( target.name.to_sym, :before_validate ) { |mem|
            # ==== password
              if mem.new?
                mem.password = mem.password.to_s.strip
                if mem.password.empty?
                  mem.errors.add(:password, "Password is required.")
                else
                  mem.salt = Member.random_string(10) # if !self.salt
                  mem.hashed_password = Member.encrypt( mem.password, mem.salt)
                end
              end

            # ==== confirm password
              if mem.new?
                mem.confirm_password = mem.confirm_password.to_s.strip
                if mem.password != mem.confirm_password
                  mem.errors.add(:password, "Password confirmation does not match password. Re-type both.")
                end
              end
        }
    end
