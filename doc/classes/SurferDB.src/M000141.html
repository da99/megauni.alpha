  def self.reset_db!
    require 'sequel/extensions/migration' 
    conn.log_info( ">>>>>>>> Starting resetting database...")
    raise ArgumentError, ":reseting the database not allowed in :production environment." if BusyConfig.production?
    Sequel::Migrator.apply( conn, Pow!('../migrations'), 0 )
    migrate_up!
    conn.log_info( ">>>>>>>> Finished resetting database.")
    
  end
