  def create # POST
    
    # ========================================================
    # Set id to zero just in case it is nil or an empty string.
    # ========================================================
    clean_room[:bundle_id] = clean_room[:bundle_id].to_i 
    # ========================================================
    
    begin
      # Create bundle if necessary
      old_bundle      = clean_room[:bundle_id] == 'new_bundle' && current_member.bundles_dataset.where(:title=>clean_room[:new_bundle_title]).first
      create_bundle   = clean_room[:bundle_id] == 'new_bundle' && old_bundle.nil?
      use_old_bundle  = old_bundle
      
      bundle_id = if create_bundle
                    new_bundle = Bundle.new
                    new_bundle.title = clean_room[:new_bundle_title]
                    current_member.add_bundle(new_bundle)
                    new_bundle.id
                  elsif use_old_bundle
                    old_bundle.id
                  else
                    clean_room[:bundle_id]
                  end
      
      # Now, let's create the check list.
      check_list = CheckList.new
      check_list.new_values_from_member(clean_room)
      current_member.add_check_list(check_list)
      success_msg!(check_list.success_msg, :redirect_to=>"/check_list/#{check_list.id}")
      
    rescue Bundle::Invalid
      error_msg!(new_bundle.error_msg)
      
    rescue CheckList::Invalid
      error_msg!(check_list.error_msg)

    end
  end
