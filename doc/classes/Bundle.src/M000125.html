  def find_validation_errors
    # === title
      self.errors.add(:title, "A title is required.") if self.title.blank?
      if new?
        duplicate_bundle = self.owner.bundles_dataset[:title=>self.title]
        self.errors.add(:title, "Already another bundle called: #{self.title}") if duplicate_bundle 
      end
  end
