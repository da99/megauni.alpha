  def find_validation_errors
    
    # === url
    self.url = self.url.to_s.strip


    # === :body
    self[:body] = self[:body].to_s.strip

    # === :title
    self[:title] = self[:title].to_s.strip
    if self.url.empty? && self[:body].empty? && self[:title].empty?
      self.errors.add(:title, "HRL, title and body can not be empty at the same time.")
    end
    
  end
