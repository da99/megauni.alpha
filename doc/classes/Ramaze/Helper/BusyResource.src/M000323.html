      def create( *raw_ids ) 

        find_instance_vars

        ancestors_are_required!

        begin
          @child = @child_profile[:record] ||= @child_model.new
          @child.update_from_editor( clean_room , current_member )
          if just_one_model?
            current_member.send( "add_#{@child_model.name_underscored}", @child)
          else
            @parent.send("add_#{@child_model.name_underscored}", @child)
          end
          load_instance_vars
          success_msg!( @child.success_msg )
        rescue @child_model::Invalid
          error_msg!( @child.error_msg )
        end

      end
