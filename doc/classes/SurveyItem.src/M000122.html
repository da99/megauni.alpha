  def ratings_summary  
    @ratings_summary ||= question.ratings_menu.map { |rating|
      rating.merge(  { :total_responses =>
                         Integer(rating[:score]) > 0  ?
                         self[:total_ratings_positive] :
                         self[:total_ratings_negative]
      })
    }
  end
