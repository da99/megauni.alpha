  def self.validate_username_format(username)
    raise(  Invalid , "Username: has to be a string."  ) unless username.instance_of?(String)
    
    username                = username.strip
    valid_username_chars    = VALID_USERNAME_FORMAT
    raise(  Invalid, "Usernames have to be between 2 and 25 characters with only the following characters:" +
                    " letters (English), numbers, underscores, periods, dashes (minus signs)."  ) unless username =~ valid_username_chars
    
    username
  end
