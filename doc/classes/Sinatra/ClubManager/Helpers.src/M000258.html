            def integerize_splat_or_captures
                raw_vals = ( params[:splat] || params[ :captures ] )
                raise "No Integers/IDs found."  unless raw_vals
                raw_vals.map { |raw_i| 
                    raw_i.split('/').map { |i|
                        Integer(i) unless i.strip.empty?
                    }
                 }.flatten.compact
            end
