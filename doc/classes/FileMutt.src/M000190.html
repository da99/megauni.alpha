  def go_fetch(&blok)
  
    @find_files.uniq!
    @not_files.uniq!
    
    @dirs.uniq!
    @not_dirs.uniq!
    
    file_patterns_used = {}
    dirs_patterns_used = {}
    
    files = []
    rejected_files = []
    dirs = []
    rejected_dirs = []
    unknown_types = []
    
    require 'find'
     
    Find.find(@wd) do |full_path|
    
      filename    = File.basename(full_path)
      dir         = File.dirname(full_path)
      folder_name = File.basename(dir)
      basedir     = dir.sub(@wd, '').gsub(/^\/|\/$/,'') # Dirname minus current working directory. 
      
            
      if FileTest.file?(full_path)
        # Check if this file is not allowed.
        unless file_allowed?(full_path)
          rejected_files << full_path
          next
        end
        
        files << full_path
        blok.call(full_path) if block_given?
      elsif FileTest.directory?(full_path)
      
        # Check if this directory is allowed.
        unless dir_allowed?(full_path)
          rejected_dirs << full_path
          Find.prune
          next
        end   
        
        dirs << full_path
        
      else
      
        unknown_types << full_path
        
      end
      
    end
    
    {:rejected_files => rejected_files, :rejected_dirs=>rejected_dirs, :files => files, :dirs => dirs, :unknown=>unknown_types }   
  end
