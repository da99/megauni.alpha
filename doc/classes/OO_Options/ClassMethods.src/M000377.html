        def oo_set(raw_title, raw_val)
            title = oo_clean(raw_title)
            if oos[title][:validate]
                oos[title][:value] = oos[title][:validate].call( raw_val )
            else
                oos[title][:value] = oo_clean_value(raw_val)
            end
        end
