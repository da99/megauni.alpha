      def update( *raw_ids ) 

        find_instance_vars
        child_is_required!

        begin
          @child.update_from_editor( clean_room , current_member )
          @child.update_it!
          @redirect_to = @child.a_href(:edit)
          success_msg!( @child.success_msg )
        rescue @child_model::Invalid
          error_msg!( @child.error_msg )
        end

      end
