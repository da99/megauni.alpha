  def send_welcome_msg
    return if self.email.blank?
    # Send a welcome email.
    subj  = "#{BusyConfig::SITE_NAME} Welcome Message"
    body = "Your account has been created."
    body << "\nThis email is to make sure you correctly typed your email address."
    body << "\nAdd #{BusyConfig::SITE_SUPPORT_EMAIL} to your address book if you ever need support."
    self.deliver_basic_msg( subj , body ) 
  end
