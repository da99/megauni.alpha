  def changes_from_editor(params, mem = nil)

    case mem
        when self
            @current_editor = mem
            @editor_may_change = [:password, :password_confirmation, :email]
        when nil
            @current_editor = :new_member if new?
        else
            if mem.is_admin?
                @current_editor = mem 
                @editor_may_change = [:password, :password_confirmation, :email]         
            end
    end # case
    
    super
  end
