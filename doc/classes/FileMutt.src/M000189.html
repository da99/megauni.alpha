  def dir_allowed?(full_path)
    raise(ArgumentError, "Error in FileMutt found.  Only valid director names are allowed in :dir_allowed?") unless FileTest.directory?(full_path)
    
    folder_name = File.basename(full_path)
    basedir     = full_path.sub(@wd, '').gsub(/^\/|\/$/,'') # Dirname minus current working directory.
    return(false) if  @not_dirs.detect { |dir_regex|  folder_name  =~ dir_regex || 
                                                       basedir      =~ dir_regex  }  
    return(false) if !@hidden_dirs && hidden?(folder_name)
    return(false) if !@dirs.empty? && !@dirs.detect {|dir_regex|   folder_name  =~ dir_regex || 
                                                                   basedir      =~ dir_regex  }  
    true
  end
