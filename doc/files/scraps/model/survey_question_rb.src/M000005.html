  def find_validation_errors
  
    # === :question_type
    self[:question_type] = Integer(self[:question_type])    
    unless VALID_QUESTION_TYPES.include?(self[:question_type])
      raise "Programmer error: Invalid question type."
    end

    # === :title
    self[:title] = self[:title].to_s.strip
    if self[:title].empty?
      self.errors.add(:title, "Title is required.")
    end
    
  end
