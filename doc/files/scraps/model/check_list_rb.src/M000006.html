  def set_success_msg(action=nil)
    case action.to_sym
      when [:create, :created]
          if self.bundle
            self.set_success_msg = "Check list, #{self[:title]}, was been created under bundle, #{self.bundle.title}."
          else
            super
          end
      else
          super
    end
  end
