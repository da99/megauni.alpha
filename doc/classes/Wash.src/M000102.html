  def self.plaintext( raw_str, *opts)
    return nil unless  raw_str

    @plaintext_allowed_options ||= [ :spaces, :tabs ]
    
    orig_kcode = $KCODE
    $KCODE = 'u'
    
    opts.each { |opt| 
      raise(ArgumentError,
            "Invalid option: #{opt.inspect}. Try #{@plaintext_allowed_options.map(&:inspect).join(', ')}"
           ) unless @plaintext_allowed_options.include?(opt)
    }

    # Save tabs if requested.
    raw_str = raw_str.gsub("\t", "&#09;") if opts.include?(:tabs)
    
    
    final_str = raw_str.split("\n").map { |line| 
                  # Normalize characters BEFORE stripping out control characters.
                  # See: http://msdn.microsoft.com/en-us/library/ms776393(VS.85).aspx
                  line.multibyte_chars.normalize.gsub( INVALID_UNICODE, '' )
                }.join("\n")

    # Save whitespace or strip.
    final_str.strip! unless opts.include?(:spaces)

    # Put back tabs by request.
    final_str.gsub!("&#09;", "\t") if opts.include?(:tabs)

    $KCODE = orig_kcode
    
    final_str.sanitize
    
  end
