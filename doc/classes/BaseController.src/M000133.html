    def route_it( opts = {} )
      raise ArgumentError, ":from is required" unless opts.has_key?(:from)
      raise ArgumentError, ":to is required" unless opts.has_key?(:to)
      from = ( opts[:from].is_a?(Symbol) ?
               R(self, opts[:from]) : 
               opts[:from].to_s
             ).gsub(  %r!^/*|/*$!, '')
      to   = ( opts[:to].is_a?( Symbol ) ? 
               R(self, opts[:to]) :
               opts[:to].to_s
             ).gsub( %r!/\\?d(?=\b)! , '/%d' )
      from_regex = %r!^/#{from}/?! 
      # Ramaze::Log.info ">>> Routing: #{from_regex}  -->   #{to} " if BusyConfig.development?
      Ramaze::Route[ from_regex ] = to
    end
