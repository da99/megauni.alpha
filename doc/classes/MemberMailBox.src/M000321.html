  def expand_text_shortcuts(raw_str)
    new_str = raw_str.to_s.strip
    @compiled_shortcuts ||= text_shortcuts.inject({}) { |m,ts| m[ /#{ts.pick('SHORTCUT')}(?=\ |$)/i ] = ts.pick('TEXT'); m}
    @compiled_shortcuts.each do | pattern, text|
      new_str = new_str.gsub(pattern, text)
    end

    new_str
  end
