  def css_classes

    class_names = [ self.class.name.underscore ]

    case self[:question_type]
      when MOST_FAV
        class_names << 'most_fav'
      when MOST_FAV_LEAST_FAV
        class_names << 'most_fav_least_fav'
      when YES_NO
        class_names << 'yes_no'
    end

    if self.items.zero?
      class_names << 'no_items'
    end
    
    class_names.join('')
    
  end
