  def self.parse_into_hash( raw_body )

      raise 'Not a string' unless raw_body.respond_to?(:each_line)
      
      body                 = plaintext(raw_body)
      parts                 = {}
      who_knows       = '' # holds unparsable text.    
      
      # Escape any text with custom escape character to prevent unwanted replacements/manipulation
      # further down during processing.
      PARSE::ESCAPES.each {|pattern, replacement|
        body = body.gsub("#{PARSE::ESCAPE_CHAR}#{pattern}", replacement)
      }
      
      
      # Start extraction process.
      curr_col                   = nil
      raw_parts                 = {}
      who_knows_arr       = []
      body.each_line { | line | 
        break if line =~ PARSE::STOP_EXTRACTION
        results =  line.scan(PARSE::FIND_COL_NAME).flatten
          
        if !results.first.blank?
          curr_col = turn_into_column_key(results.first) # Sanitize key name.
          
          # If column name is a "combinable" column (REMOVE TAG, ADD TAG, etc.), combine the value
          # if possible. If not, then create a  new Hash key/value.
          combined_col, combined_pattern = PARSE::COMBINE_COLS.detect { |col_name, pattern|  curr_col =~ pattern}
          if combined_col
            curr_col = combined_col
            raw_parts[curr_col] ||= ''
            raw_parts[curr_col] << " #{results.last} "
          else
            raw_parts[curr_col] ||= results.last      # Save everything after the first DIVIDER_CHAR        
          end
          
        elsif curr_col
          raw_parts[curr_col] << line
        elsif !curr_col
          who_knows_arr << line
        else
          raw_parts[curr_col] = results.last
        end 
      }
      
      who_knows = who_knows_arr.join("\n")
      
      # ------------------------------------------------------------------------------------------------
      
  #    # Extract any unknown text from the start of the ENTIRE String,
  #    # not just starts of lines.
  #    raw_who_knows = body.scan( PARSE::UNKNOWN_BEGINNING_TEXT ).flatten.first
  #    unless raw_who_knows.blank?
  #      who_knows << raw_who_knows
  #    end
      
  #    # Extract columns and values.
  #    # However, we first extract the text before the delimiter (PARSE::END_EXTRACTION),
  #    #   (The delimiter is usually the string '#" on a single line.)
  #    raw_parts = if body =~ PARSE::END_EXTRACTION
  #                         $1.scan(PARSE::WITH_EXTRACTOR)
  #                      else
  #                        body.scan(PARSE::WITH_EXTRACTOR)
  #                      end
      
      # We loop through each of the raw results to 
      # get the finished {'key'=>'value'} hash (sanitized, stripped, etc.)
      unless raw_parts.empty?
    
        # Take text matches and put them as a {'key'=>'value'} hash.
        raw_parts.each { |key, val| 
        
          key_as_col = turn_into_column_key(key)
          
          # UNescape previous escaped text.
          PARSE::ESCAPES.each {|pattern, replacement|
            val = val.gsub(replacement, pattern)
          }
          
          # Combine certain values if needed (e.g. multiple use of Remove Tag).
          # Also, standardize the column value. (e.g. 'remOve tAg' is change to 'REMOVE TAGS').
          PARSE::COMBINE_COLS.each do |col, col_pattern|
            if key_as_col =~ col_pattern
              key_as_col   = col
              val                 =  "#{parts[key_as_col]} #{val}"            
            end
          end
          
          parts[key_as_col]  = val.rstrip
          break if $1;
        }
      end
      
      # Take out any disallowed column names.
      parts.delete_if do |key, val|
        delete_it = false
        PARSE::SPECIAL_COLS.each do |special_col|
          if special_col =~ key
            who_knows << "\n#{key} #{PARSE::DIVIDER_CHAR}\n#{val}"
            delete_it = true
          end
        end
        delete_it
      end
      
      # Make entire body (raw_body after it has been cleaned)
      # unparsable text if no parts were found.
      who_knows = body if parts.blank?
      
      [who_knows, parts]
  end
