    def put(*args, &old_proc)
        raise "PROC IS MISSING FOR CUSTOM :put: #{args.inspect}" unless old_proc
        new_action_name, perm_level, path =  args
        props = {  :action => new_action_name ,
            :path=> path || generic_path(new_action_name), 
            :http_verb=>__method_name_sym__, 
            :perm_level=>perm_level,
            :controller=>controller_name
        }           
        validate_props!(props)   
        eval( generic_action( props , old_proc.to_ruby.to_s + '.call(*args)' ) , self.original_scope )
    end
