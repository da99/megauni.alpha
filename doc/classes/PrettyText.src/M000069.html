  def self.center_line(raw_text, char_decorator = ' ')
    new_text = raw_text.strip
    half_size       =  ( ( PRINT::LINE_SIZE / 2) - (new_text.size/2) ) 
    return new_text if half_size < 0
    compensate =  ( PRINT::LINE_SIZE - ( new_text.size + (half_size *2) ) )
    return new_text if (half_size  + compensate) < 0
    ( char_decorator * (half_size  + compensate) )  +  new_text + ( char_decorator  * (half_size))
  end
