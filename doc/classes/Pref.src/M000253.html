  def find_validation_errors
    # === title
      self.filename = Wash.filename(self.filename)
      raise ArgumentError, "#{filename} is required." if self.filename.blank?
    
    # === model_class_name
      raise ArgumentError, "#{model_class_name} is required." if self.model_class_name.blank?
      
    # === model_id
    raise ArgumentError, "#{model_id} is required." if self.model_id.to_i.zero?
      
    # === body
      self.errors.add(:body, "A value is required.") if self.body.blank?
  end
