  def find_validation_errors
    # === path
      self.path = Wash.path( self.path )
      self.path = BusyConfig.publicize_path(self.path) unless dir_exists?
      if new?
        df = DiskFile[:path=>self.path]
        df.destroy if df   
      end
      raise ArgumentError, "Path is not in public directory: #{self.path}" unless public_dir?
      raise ArgumentError, "File exists on disk, but not in db. Delete it manually if you want to overwrite it." if possible_important_file?
      raise ArgumentError, "Directory does not exist." unless dir_exists?
      raise ArgumentError, "Filename needs a proper 2-4 character extension." unless has_file_extension?
    
    # === content
      self.content = self.content.strip
      raise ArgumentError, "No content specified for: #{self.path}" if self.content.blank?
    
  end
