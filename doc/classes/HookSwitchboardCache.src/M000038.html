  def self.do_this( action_as_sym, item)
    # Validate model names and cache results so we don't have to do it next time..
    @model_names_validated ||= begin
        __switch__.keys.each { |sym_name| 
          raise InvalidClassName, "Invalid model name: #{sym_name.inspect}" unless Object.const_defined?(sym_name)
        }    
    end
    
    hooks = __switch__[item.class.name.to_sym]

    if hooks && hooks[action_as_sym]
      BusyConfig.dev_log_it "Executing: #{item.class.name.inspect} for #{action_as_sym.inspect}"
      hooks[action_as_sym].each { |stuff|
        stuff.call(item)
      }
    #else
    #  BusyConfig.dev_log_it "^^^ No action found. Not going to do anything: #{item.class.name.inspect} for #{action_as_sym.inspect}"
    end

  end
