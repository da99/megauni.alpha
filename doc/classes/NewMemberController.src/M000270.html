  def create # POST create
    
    session.clear
    
    begin
      m = Member.new
      m.username  = clean_room['username']
      m.password  = clean_room['password']
      m.confirm_password = clean_room['confirm_password']
      m.email     = clean_room['email']    
      m.create_it!
      
      flash[:success_msg]=m.success_msg
      store_member_to_session(m)
      redirect('/admin')
      
    rescue Member::Invalid
      session.clear
      flash[:error_msg] =  m.error_msg
      flash[:username]  = clean_room['username']
               
      redirect('/signup', :status => 302)
    end
    
  end
