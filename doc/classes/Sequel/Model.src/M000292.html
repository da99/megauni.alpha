  def update_from_editor(raw_hash,  editor = nil)
    current_editor = editor
    new_vals = raw_hash.dup # Just in case we delete any values,
                                            # we do not want to affect the original Hash
                                            # if :raw_hash is passed by reference.
                            
    editable_fields = self.class.oo_get(:editor_may_change).instance_of?(Proc) ?
                                    self.class.oo_get(:editor_may_change).call(self) :
                                    self.class.oo_get(:editor_may_change) ;

    (new_vals.keys.map {|field| field.to_s} - editable_fields.map {|field| field.to_s}).each { |banned_field|
      raise FieldNotAllowedToBeUpdatedByOwner, "method #{banned_field}= is not allowed when using input from user."
    }
    
    editable_fields.each { |fieldname|
      if new_vals.has_key?(fieldname.to_s)
        self[fieldname.to_sym] = new_vals[fieldname.to_s]
      end

      if new_vals.has_key?(fieldname.to_sym)
        self[fieldname.to_sym] = new_vals[fieldname.to_sym]
      end
    }
  end
