    def self.validate_hash_keys!( submitted, valid_opts )
        last_call = caller[-1] || caller.last || "(:caller could not determine line number)"
        invalid_fields = ( submitted.respond_to?(:keys) ? submitted.keys : submitted ) -  valid_opts
        raise ArgumentError, "Invalid fields: #{invalid_fields.join(', ')}. Look at line: #{last_call}" if !invalid_fields.empty?
    end
