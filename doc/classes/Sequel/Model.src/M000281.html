  def wash_language_fields
    schema = self.class.db_schema
    
    self.class.columns.select { |col| col.to_s["language"] }.each do |col|
      self[col] ||= Language.all.first
      self.errors.add(col, "Language is not accepted.") unless Language.all.include?(self[col])
    end
    
  end
