  def validate_it!
    errors.clear
    strip_and_utc_everything
    find_validation_errors
    
    # === set error_msg
    self.error_msg = begin
        all_errors = self.errors.values.flatten
        all_errors.size === 1 ?
            all_errors.first  :
            '* ' + all_errors.join("\n* ")   
    end
    
    # === raise Exception if necessary
    raise( Invalid, error_msg ) unless errors.empty? 
    
    true
  end
