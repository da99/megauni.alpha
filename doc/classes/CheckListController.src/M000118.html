  def delete( id = 0 ) # DELETE
    begin
      @check_list = current_member.check_lists_dataset.first_or_raise( :id=>clean_room[:id] )
      @check_list.destroy    
      success_msg!(@check_list.success_msg, :redirect_to=>'/work')
    rescue CheckList::NoRecordFound
      success_msg!("#{@check_list.title} was successfully deleted.")
    end
  end
