  def find_validation_errors

    # === :title
    self[:title] = self[:title].to_s.strip
    if self[:title].empty?
      self.errors.add(:title, 'A title is a required.')
    end

    # === :status
    self[:status] = Integer(self[:status])
    unless VALID_STATUSES.include?(self[:status])
      raise ArgumentError, "Invalid status."
    end

    # === :bundle_id - Check survey owner has the same bundle owner.
    if self.bundle_dataset.first
      if self.bundle_dataset.first.owner != self.owner_dataset.first
        raise InvalidOwner, "Owner does not match bundle owner: owner_id: #{self.owner_id}, bundle owner_id: #{self.bundle_dataset.first.owner_id}"
      end
    end
    
  end
