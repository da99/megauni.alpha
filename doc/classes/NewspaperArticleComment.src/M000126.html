  def changes_from_editor( params, mem = nil )
    if new?
        @current_editor  = mem || :not_signed_in
        @editable_by_editor = [:author_name, :body]
    elsif mem && [author, article.author, article.newspaper.owner].include?(mem)
        @current_editor = mem 
        @editable_by_editor = [:trashed]
    end
    super
  end
