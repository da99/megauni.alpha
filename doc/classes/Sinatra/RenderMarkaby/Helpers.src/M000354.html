            def render_mab( opts = {} )
            
                response['Content-Type'] = 'text/html; charset=utf-8'
                response['Accept-Charset'] = 'utf-8'   
                
                # Find template. ==================================================
                skins_dir = File.join( self.views, 'jinx' )
                action_template =  File.join( skins_dir, self.template_name+ '.mab'  )
                layout_template =  File.join( skins_dir,  'layout.mab' )
                
                # ======= See if we need to use a layout.
                is_partial = self.template_name =~ /^__/ 
                use_layout = !is_partial 
                                
                # Setup instance variables for Markaby. ==========================
                iv_hash = instance_variables.inject({}) do |m, iv|
                    key = iv.gsub('@', '').to_sym
                    m[key] = instance_variable_get(iv)
                    m
                end

                # Setup Markaby Builder. =========================================
                mab  = Markaby::Builder.new(iv_hash, helpers = self)

                # Add dirs of templates to Markaby to help find partials. ========
                mab.add_template_root( skins_dir )

                # =================================================================
                # Determine if a layout is required.
                if !use_layout 
                    BusyConfig.dev_log_it "Rendering Template w/o Layout: #{action_template}" 
                    return mab.capture { eval(File.read(action_template)) }
                end

                #  =================================================================
                # Grab & Render the Markaby content for current action.
                BusyConfig.dev_log_it "Rendering Markaby: #{action_template}" 
                mab.save_to('the_content') {
                    eval( File.read(action_template) , nil, action_template )
                }

                #  =================================================================
                # Update iv_hash for :the_content.
                # Grab & Render Markaby content for layout.
                BusyConfig.dev_log_it "Rendering Markaby layout: #{layout_template}" 
                mab.instance_eval(  File.read( layout_template ),  layout_teplate  ).to_s
              
            end
