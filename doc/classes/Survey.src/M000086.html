  def css_classes

    class_names = [ self.class.name ]
    
    class_names << case self[:status]
      when DRAFT
         'survey_draft'
      when LIVE
        'survey_live'
      when CLOSED
        'survey_closed'
    end
    
    if self[:last_response_at].nil?
      class_names << 'survey_no_responses'
    end
    
    class_names.join(' ')
    
  end
