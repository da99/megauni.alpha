        def oo_create( raw_title, raw_val, raw_what_it_does = '', validate_lambda = nil)
            
            title = oo_clean_title(raw_title);
            what_it_does = raw_what_it_does.to_s.strip
            meth_valid = "#{title.underscore}_oo_validate"
            
            if validate_lambda 
                val = validate_lambda .call( raw_val )
            else
                what_it_does = "No validation lambda found.\n\n" + what_it_does
                val = oo_clean_value(raw_val) 
            end
            
            @oo_options[ o_name ] = { :body=>val, 
                                                            :description=>what_it_does, 
                                                            :validate=>validate_lambda 
                                                          }
        end
