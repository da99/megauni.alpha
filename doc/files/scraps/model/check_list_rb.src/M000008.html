  def done?
    # Make sure we are dealing with Integers
    self.total_items  = Integer(self.total_items)
    self.done_items   = Integer(self.done_items)

    # Put the comparisons into words to avoid confusion.
    no_items    = self.total_items < 1
    no_match    = self[:total_items] != self[:done_items]
    we_have_a_match  = self[:total_items].equal?( self[:done_items] )

    # Let's return some values.
    return false if no_items
    return false if no_match
    
    return true  if we_have_a_match
    
    false # <<<< The default.
    
  end
