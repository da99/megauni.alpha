  def set_success_msg(action=nil)
    return super unless [:create, :created].include?(action)
    
    self.success_msg =  "Your account has been created. \n" +
                        "Contact #{BusyConfig::SITE_SUPPORT_EMAIL} "
    if self.email.blank?
      self.success_msg << " if you have further questions. "
    else
      self.success_msg << " if you did not receive your personalized instructions at your email address."
    end
  end
