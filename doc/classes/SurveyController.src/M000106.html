  def create #  POST
    require_post!

    begin
      new_survey = Survey.new
      new_survey.new_values_from_member(clean_room)
      current_member.add_survey(new_survey)
      success_msg!( new_survey.success_msg, :redirect_to=>"/survey/edit/#{new_survey[:id]}" )
    rescue Survey::Invalid
      error_msg!(new_survey.error_msg)
    end
  end
