  def deliver_msg(raw_subject, raw_body)
    return nil if self.email.blank?
    
    BusyConfig.deliver_msg( :to =>member.email, 
                            :from=>"\"#{BusyConfig::SITE_NAME} Computer\" <#{member.mail_box}>", 
                            :subject=>raw_subject, 
                            :body=>raw_body )    
    
  end
