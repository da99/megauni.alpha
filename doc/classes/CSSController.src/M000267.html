  def grab_and_render( raw_layout_dir, raw_filename )
    
    response['Content-Type'] = 'text/css' 

    css_dir       = LAYOUTS_DIR / raw_layout_dir / 'css'
    sass_template = css_dir / (raw_filename + '.sass')

    if BusyConfig.development?
      raise( SASSFileNotFound, "Not found: #{sass_template}" ) unless sass_template.file?
    end

    engine = Sass::Engine.new(
                sass_template.read, 
                :load_paths=>[  
                  css_dir,
                  COMPASS_SASS_DIR,
                  SASS_960_DIR
                ]
    )

    engine.render 
    
  end
