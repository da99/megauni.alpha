  def find_validation_errors
    # === title
    require_string! :title, "A title is always required."

    # === layout_id
    self[:layout_id]  = self[:layout_id].to_i
    raise "Invalid layout id" unless VALID_LAYOUT_IDS.include?(self[:layout_id])

    # === css_layout_id
    self[:css_layout_id]    = self[:css_layout_id].to_i
    raise "Invalid css layout id" unless VALID_CSS_LAYOUT_IDS.include?(self[:css_layout_id])
  end
