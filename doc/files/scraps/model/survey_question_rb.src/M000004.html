  def ratings_menu
    item_count = items_dataset.count
    self[:question_type] = Integer(self[:question_type])
    real_question_type = item_count === 1 ?
                         YES_NO :
                         ( [MOST_FAV, MOST_FAV_LEAST_FAV].include?(self[:question_type]) ?
                            self[:question_type] :
                            MOST_FAV )
    RATINGS_MENU[ real_question_type ]
  end
