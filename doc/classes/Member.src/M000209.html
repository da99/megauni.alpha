  def self.most_thanked(raw_this_many = 10)
    this_many = Integer(raw_this_many)
    query {
      where(:username=> ThankYouStat.query {
                         select(:recipient_username)
                         order_by(:score.desc)
                         limit(this_many)
                        }
      )
      order_by(:created_at.desc)
    }
  end
