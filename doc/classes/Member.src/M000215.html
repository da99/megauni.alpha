  def paper_trail_update!(action= nil)
    return if @old_this.blank?
    @paper_trail                = PaperTrai.new
    @paper_trail.body           = PaperTrail.differ_hash( old_vals, self.this.first  ) 
    return if @paper_trail.body.blank?
        
    @paper_trail.owner_id = self[:id]
    @paper_trail.model_class_name     ||= self.class.name
    @paper_trail.model_id       ||= self.id
    @paper_trail.action         ||= action
    @paper_trail.save
  end
