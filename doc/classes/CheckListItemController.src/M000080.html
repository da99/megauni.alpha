  def update( check_list_id = 0, item_id = 0 ) # POST
    begin
      @check_list = current_member.check_lists_dataset.first_or_raise( :id=>clean_room[:id])
      @item = @check_list.items_dataset.first_or_raise( :id=>clean_room[:id] )
      @item.new_values_from_member(clean_room)
      @item.save
      success_msg!(@item.success_msg, :partial=>'__ajax_check_list_item''__ajax_check_list_item')
    rescue CheckListItem::NoRecordFound
      error_msg!("Item not found. Contact support.")
    rescue CheckList::NoRecordFound
      error_msg!("Check list not found. Contact support.")
    rescue CheckListItem::Invalid
    end
  end
