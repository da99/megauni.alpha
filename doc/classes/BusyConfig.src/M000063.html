  def self.timezones
    return @us_zones if @us_zones
    
    require 'tzinfo'
    countries = [ ['JP', 'Nippon' ], ['US', 'U.S.A.'] , ['CA', 'Canada'], ['GB', 'United Kingdom'] , ['IE', 'Ireland'], ['AU', 'Australia'], ['NZ', 'New Zealand'] ]
    
    @us_zones = {}
    
    countries.sort.each do |c|
      @us_zones[c[1]] = TZInfo::Country.get(c[0]).zones.sort {|x, y| y.current_period.utc_total_offset <=> x.current_period.utc_total_offset}.map { |t| 
                                      num_sign = t.current_period.utc_total_offset > 0 ? '+' : ''
                                      [ t.identifier,  "(GMT#{num_sign}#{t.current_period.utc_total_offset/60/60} Hrs.) #{t.name.split('/')[1,10].join('/').gsub('_', ' ')}" ]
                                    }
    end
    @us_zones['World'] = [ ['UTC', 'Universal Time (GMT)'] ]
    
    @us_zones

  end
