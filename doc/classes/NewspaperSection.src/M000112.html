  def find_validation_errors
    # === :title
    require_string! :title, "You left out the name of this section."

    # === :placement
    self[:placement_id] = Integer(self[:placement_id])
    raise "Invalid placement for newspaper section." unless Newspaper::VALID_PLACEMENT_IDS.include?(self[:placement_id])

  end
