        def guard_it_for( method_type, perm_level, path, raw_controller_name, raw_action_name, raw_is_partial = :no_partial, &custom_action )
            
            # === Validate partial option.
            raise "INVALID PARTIAL OPTIONAL: #{raw_is_partial}" unless [:no_partial, :partial, :no_template].include?(raw_is_partial)
            is_partial = raw_is_partial === :no_partial ? false : true
            has_template = raw_is_partial === :no_template ? false : true
            
            # Set names of controller and action.
            temp_controller_name = raw_controller_name.to_s.strip.camelize.to_sym
            temp_action_name = raw_action_name.to_s.strip.upcase.to_sym
            
            
            # === Validate HTTP verb.
            if !self.valid_http_resourceful_actions.include?(method_type)
                raise "INVALID HTTP METHOD: #{method_type.inspect }"
            end
            
            # === Page must not exist.
            cache_name = path.respond_to?(:strip) ? path.strip : path # Previous way ==> path.to_s + method_type.to_s
            if self.pages_made_cache[ cache_name ] 
                raise "PATH ALREADY TAKEN: #{:path} in #{self.pages_made_cache[ cache_name ][:controller_name]} " + 
                            "#{self.pages_made_cache[ cache_name ][:action_name]}"
            elsif( existing_controller_action = self.pages_made_cache.value.detect {   |vals| 
                        vals[:controller_name].eql?(temp_controller_name) && vals[:action_name].eql?(temp_action_name) })
                 raise "CONTROLLER NAME AND ACTION NAME TAKEN: #{temp_controller_name.inspect} #{temp_action_name.inspect}"
            end
                                                                           
            self.pages_made_cache[ cache_name ] = {  :perm_level=>perm_level, 
                    :path => path, 
                    :action =>temp_action_name,
                    :is_partial => is_partial, 
                    :controller  => controller_name,
                    :template => [ (is_partial ? '__' : nil),  controller_name.to_s.underscore,  temp_action_name.to_s.underscore  ].compact.join('_'),
                    :orm_class => Object.const_defined?( current_action[:controller] ) ? 
                                                Object.const_get( current_action[:controller]  ) :
                                                nil
            }
            
            
            # === E.g.:    
            #  get( "/" ) { 
            #      check_creditials!(STRANGER) 
            #   }
            eval( %~
                #{ method_type.to_s.underscore  }( #{path.inspect} ) do
                    describe_action #{path.inspect}
                    
                    #{  strigify_proc( custom_action )  }
                    
                    #{ has_template ? "render_mab" : '' } 
                end
            ~ )
            
        end
