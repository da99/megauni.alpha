        def oos
            @oo_options ||= begin
                new_o = {}
                sequel_ancestors = ancestors.select { |m|   m.respond_to?(:oos) && m != self   }
                sequel_ancestors.each  { | anc| 
                                                            if anc.respond_to?(:oos) && anc.oos.is_a?(Hash)
                                                                anc.oos.each { |k,v|
                                                                    new_o[k] = v.dup
                                                                }
                                                            end
                                                        } 
                new_o                                                                  
            end
            
            @oo_options
        end
