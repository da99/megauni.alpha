  def self.ramaze_options
    
    raise EnvironmentNotSet, "This method can not be called until environment has been set." unless self.environment
    raise AppSetNotToRun, "Can't run the app while using Rake." if Object.const_defined?(:Rake)
    raise AppSetNotToRun, "Can't run the app while testing." if test?
    
    case self.environment
      when ENV_DEV
        {
          :adapter => :thin,
          :port => 4567,
          :sourcereload => false
        }
      when ENV_PROD
        {
          :adapter => :thin,
          :port => 9123,
          :sourcereload => false,
          :boring => [/\.css$/, /\.js$/, /\.img$/] # from: http://justkez.com/deploying-ramaze-to-production/
        }
    end
    
  end
