  def clone_it!
    clone = CheckList.new
    clone.owner_id    = self[:owner_id]
    clone.title       = self[:title]
    clone.body        = self[:body]
    clone.template_id = self[:id]
    clone.bundle_id   = 0
    clone.save

    self[:total_clones] = self.clones_dataset.count
    save
    
    self.items.each { |old_item| 
      old_item.clone_it!( clone )
    }
    
    clone
  end
