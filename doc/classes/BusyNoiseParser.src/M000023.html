  def self.parse_tags(tag_str_or_arr)
    raise ArgumentError, 'Argument needs to be a String or an Array.' if !(  tag_str_or_arr.to_s == tag_str_or_arr  || tag_str_or_arr.instance_of?(Array) )
    
    tag_str = if tag_str_or_arr.instance_of?(Array)
                      tag_str_or_arr.flatten.compact.uniq.join(' ')
                    else 
                      tag_str_or_arr
                    end
    tags = tag_str.split("\n").map { |tag| 
                                                    tag.strip.blank? ? nil : tag.split(' ').map { |t| 
                                                                                                                      new_tag = __sanitize_tag__(t); 
                                                                                                                      new_tag.blank? ? nil : new_tag 
                                                    } }.flatten.uniq.compact
    tags
  end
