  def teammate_profile
    profile = newspaper.teammates_dataset.where(:member_id=>self[:author_id]).first
    if profile
      { 
        :title => profile[:title], 
        :description => profile[:description]
      }
    else
      if author.eql?( newspaper.owner )
        {
          :title=> 'Editor/Owner',
          :description=>"Owner and Editor of, #{newspaper[:title]}"
        }
      else
        {
          :title=>"Profile Deleted.",
          :description=>"No profile for this author was found."
        }
      end
    end
  end
