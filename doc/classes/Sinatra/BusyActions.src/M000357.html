        def busy_action_for_class_exists?( raw_class_name, action_name )
            class_name = raw_class_name.to_sym
            busy_action_validate! action_name
            self.busy_action_cache.has_key?(class_name) &&
                self.busy_action_cache[class_name].include?(action_name)
         end
