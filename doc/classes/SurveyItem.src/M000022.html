  def responses_changed!

    # Setup dynamic aggregates
    response_stats = responses_dataset.query {
      select(:id, :score, :COUNT.sql_function(:id).as(:total_responses))
      where( :item_id => self[:id])
      group(:score)
    }.all
  
    # Find the new value of :total_score
    # using dynamic aggregates.
    self[:total_score] = response_stats.inject(0) { |memo,resp|
      memo += ( Integer(resp[:score]) * Integer(resp[:total_responses]) )
      memo
    }

    # Find the new values for :total_ratings_positive
    # and :total_ratings_negative 
    # using dynamic aggregates.
    response_stats.each { |stat|
      Integer(stat[:score]) > 0 ?
        self[:total_ratings_positive] = stat_record[:total_responses] :
        self[:total_ratings_negative] = stat_record[:total_responses]
    }
    
    save    
  end
