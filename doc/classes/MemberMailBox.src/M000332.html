  def mail_box_owner_or_not!  target_mail_box
  
    # Is the mail box for member open?
    raise ERRORS::MailBoxIsClosed unless self.mail_box_open
  
    return true if target_mail_box.downcase.eql?( self.mail_box.downcase )
        
    self.mail_box_open = false
    save
    add_paper_trail( PaperTrail.new(  
                                        :model_name=>self.class.name, 
                                        :model_id=>self.id, 
                                        :action=>'UPDATE' , 
                                        :body=>"Mail box closed because of probable guessing attempt using: #{target_mail_box}"
                      ) )
    

    raise ERRORS::DoesNotOwnMailBox
  end
