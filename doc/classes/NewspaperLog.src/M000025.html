  def item=(new_item)

    self.item_class_as_int = self.class.const_get( new_item.class.to_s.sub('Newspaper','').upcase )
    self.item_id = new_item[:id]
  
    case new_item
    
      when NewspaperArticle
        self.title = "New Article: #{new_item[:title]}"
        self.body  = "by: #{new_item.by_line_name}"
        
      when NewspaperSection
        self.title = "New Section: #{new_item[:title]}"
        self.body  = new_item[:description]
        
      when NewspaperTeammate
        self.title = "New Teammate: #{new_item.member.username}"
        self.body  = "Role: #{new_item.title}"
        
      else
        raise "Unknown class for NewspaperLog: #{new_item.class}"
        
    end # === case new_item.class
    
  end
