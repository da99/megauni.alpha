  def self.connect()
    
    case BusyConfig.environment
      when BusyConfig::ENV_TEST, BusyConfig::ENV_DEV
          # -----------------------------------------------------------------------------------------------
          # Setup logger.
          # -----------------------------------------------------------------------------------------------
          require 'logger'
          test_file_path = Pow!('../../sequel_log.txt')
          File.delete(test_file_path) if File.exists?(test_file_path)          
          new_logger =  Logger.new(test_file_path)
          # -----------------------------------------------------------------------------------------------
          @db_conn = Sequel.connect( BusyConfig.db_connection_string  , {:loggers=> [new_logger]}  )
          
          
      when BusyConfig::ENV_PROD
          @db_conn = Sequel.connect( BusyConfig.db_connection_string )         
      else
        raise ArgumentError, "#{new_env.inspect} - is not a valid environment for SurferDB"
    end
    
  end
