        def oo_clean(val_type, raw_val)
            raise "INVALID OPTION: #{val_type}" unless [:value, :title].include?( val_type )
            val = case raw_val
                when Array
                    raw_val.map { |val| oo_clean(:value, val) }
                when Hash
                    raw_val.inject( {} ) { |m, kv|  m[kv.first] =oo_clean( :value,  kv.last ); m }
                when String
                    raw_val.strip
                else
                    raw_val
            end
            
            # === Special treatmeant if this is a title.
            if val_type == :title
                val = raw_val.to_s.strip
                raise "INVALID TITLE FOR OO OPTION: #{val.inspect}" if val.empty?                
            end

            
            val
        end
