  def find_validation_errors
    # === title
      self.errors.add(:title, "Either a title or body is required.") if self[:title].blank? && self[:body].blank?

    # === mark
      raise ArgumentError, "Invalid value for :mark: #{self[:mark].inspect}" unless ALLOWED_MARKS.include?(self[:mark])
      
  end
