  def create(raw_survey_id) # POST
    
    require_post!
    survey_id = Integer(raw_survey_id)
    
    begin # Find the survey.
      survey = Survey.first_or_raise(:id=>survey_id)
    rescue Sequel::Model::NoRecordFound
      error_msg!("Survey not found. It might have been deleted while you were filling it out.")
    end
    
    # Add values to response.
    survey_response = SurveyResponse.new
    survey_response.owner_name  = clean_room['name']
    survey_response.item_id     = clean_room['item_id']
    survey_response.rating      = clean_room['rating']
    survey_response.ip_address  = request.env['REMOTE_ADDR']
    survey_response.owner_id    = current_member[:id] if logged_in?

    begin # Save response.
      survey.add_response(survey_response)
      # Send back success message.
      success_msg!(survey_response.success_msg, :redirect_to=>"/survey/#{survey[:id]}/results")
    rescue SurveyResponse::Invalid
      error_msg!(survey_response.error_msg)
    end
    
  end
