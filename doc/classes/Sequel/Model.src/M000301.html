  def require_string! raw_field_name, raw_error_msg  = nil
    field_name = raw_field_name.to_sym
    strip!  field_name
    error_msg  = ( raw_error_msg || "A #{field_name} is required." ).strip
    
    self.errors.add( field_name, error_msg ) if self[field_name].empty?
  end
