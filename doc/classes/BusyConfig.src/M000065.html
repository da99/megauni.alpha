  def self.deliver_letter( raw_letter )
    
    require Pow!('../lib/pony/lib/pony')
        
    letter = { 
      :to       => Wash.plaintext( raw_letter[:to] ),
      :from     => Wash.plaintext( raw_letter[:from] ),
      :subject  => Wash.plaintext( raw_letter[:subject] ),
      :body     => Wash.plaintext( raw_letter[:body], :spaces, :tabs )
    }
    
    if self.test?
      return letter 
    end
    
    if self.development?
      Ramaze::Log.warn(letter.inspect)
      return letter
    end
    
    Pony.mail(letter)
    
  end
