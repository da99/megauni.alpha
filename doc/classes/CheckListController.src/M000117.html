  def update( id = 0 ) # PUT
    begin
      @check_list = current_member.check_lists_dataset.first_or_raise( :id=>clean_room[:id] )
      @check_list.new_values_from_member( clean_room )
      @check_list.save
      
      success_msg!(@check_list.success_msg, :partial=>"__ajax_check_list""__ajax_check_list")
    rescue CheckList::NoRecordFound
      error_msg!("No check list found.")
    rescue CheckList::Invalid
      error_msg!(@check_list.error_msg)
    end
  end
