  def self.deliver_msg_to_support( raw_subject, *args )
    require 'yaml'

    raw_body = args.map { |v| 
                          v = [v.message, v.backtrace] if v.respond_to?(:backtrace) # i.e. an Exception?
                          YAML::dump v
                        }.join("\n\n------------------------------------------------------------------------------------------\n\n")
                        
    deliver_letter( :to=>SITE_SUPPORT_EMAIL, :from=>SITE_SUPPORT_EMAIL, :subject=>raw_subject, :body=>raw_body)
  end
