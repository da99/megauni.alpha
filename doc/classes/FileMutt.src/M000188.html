  def file_allowed?(full_path)
    filename = File.basename(full_path)
    return(false) if !@find_files.empty? && !@find_files.detect { |find_regex| filename =~  find_regex}
    return(false) if @not_files.detect { |find_regex| filename =~ find_regex } 
    return(false) if !@hidden_files && hidden?(filename)
    true
  end
