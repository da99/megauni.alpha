  def find_validation_errors
    # === :name
    self[:name] = self[:name].to_s.strip

    # === :score
    self[:score] = Integer(self[:score])
    unless SurveyQuestion::VALID_SCORES.include?(self[:score])
      raise "Programmer error. Invalid score."
    end

    # === :ip_address
    unless self[:ip_address]
      self.errors.add(:ip_address, 'Can not be saved. Programmer error. Contact programmer with code: survey-response-001')
    end
    
  end
