         def busy_action_for_class_validate!( raw_class_name, action_name )
            class_name = raw_class_name.to_sym
            if !busy_action_for_class_exists?( class_name, action_name )
                raise "Action, #{action_name.inspect}, does not exist for, #{class_name.inspect}"
            end
         end
