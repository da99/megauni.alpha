  def self.use_env!( new_env )
    raise EnvironmentAlreadySet, "Environment already set to: #{@env.inspect}" if @env
    
    if new_env.eql?(:guess)
    
      specs_are_running   = Object.const_defined?(:Ramaze) && ( Ramaze.const_defined?(:SpecLogger) || Object.const_defined?(:Bacon) )
         
      require Pow!( 'secret_closet' )
      
      new_env = ( SecretCloset.production? && ENV_PROD ) ||
                ( specs_are_running  &&  ENV_TEST ) ||
                ENV_DEV
                
    end

    raise ArgumentError, "'#{new_env.inspect}' is not a valid environment." unless ALLOWED_ENVS.include?(new_env)

    @env = new_env
    
    case new_env
      when ENV_DEV
        if Kernel.const_defined?(:Ramaze)
          Ramaze::Log.info "Using default #{new_env.inspect} environment."  
          Ramaze::Log.loggers.each{|l| l.log_levels << :dev }
        end
      when ENV_TEST
         # Nothing so far.
      when ENV_PROD
          setup_for_shared_host!
    end

    return @env
    
  end
