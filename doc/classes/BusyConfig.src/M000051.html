  def self.kcode_workaround( &blok ) 
    old_kcode = $KCODE
    results   = nil
    
    begin
      results = yield
    rescue RegexpError
      raise $! unless $!.message =~ /too short multibyte code string/
      $KCODE = 'NONE'
      results = yield
      $KCODE = old_kcode
    end

    results
  end
