        def oo_clean_value(raw_val)
            case raw_val
                when Array
                    raw_val.map { |val| oo_clean_value(val) }
                when Hash
                    raw_val.inject( {} ) { |m, kv|  m[kv.first] =oo_clean_value( kv.last ); m }
                when String
                    raw_val.strip
                else
                    raw_val
            end
        end
