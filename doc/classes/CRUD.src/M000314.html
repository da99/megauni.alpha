  def create_it!

    raise "RECORD IS NOT NEW: Can not use :create_it!" if !new?

    self.created_at = Time.now.utc if self.columns.include?(:created_at)

    validate_it!

    hook_it(:save){ 
        hook_it(:create) { 
          save
        }
    }
    "#{self.class.english_name.capitalize} has been saved."
  end
