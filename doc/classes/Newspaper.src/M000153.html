  def a_href(raw_action=nil)
    parts = raw_action.to_s.split('_')
    action = parts.shift.to_sym
    model = parts.join('_')


    if !['article', 'section'].include?(model)
      return super
    end

    new_model = Kernel.const_get("Newspaper#{model.camelize}").new
    new_model[:newspaper_id] = self[:id]
    new_model.a_href(action)
    
  end
