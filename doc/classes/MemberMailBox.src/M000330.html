  def self.attach_owner_to_letter( letter )
    s_addr = letter[:sender]
    f_addr = letter[:from]
    
    sender_mem   = Member.first(:email=>s_addr)
    from_mem       = Member.first(:email=>f_addr)
    new_alias             = nil

    # Check to see if both :sender and :from 
    # are the same or different.
    member = if sender_mem && sender_mem.eql?( from_mem )
                        sender_mem
                    elsif sender_mem && !from_mem
                        new_alias = f_addr unless sender_mem.pending_email_aliases_dataset.first(:email=>f_addr) && letter[:subject] =~ EmailAlias::CONFIRMING_EMAIL
                        sender_mem
                    elsif !sender_mem && from_mem
                        new_alias = s_addr unless from_mem.pending_email_aliases_dataset.first(:email=>s_addr) && letter[:subject] =~ EmailAlias::CONFIRMING_EMAIL
                        from_mem
                    else 
                      nil                        
                    end
                    
    unless member
      # If no Member is found based on primary email and sender/from addresses, 
      # Then let's search aliases.
      sender_ea = EmailAlias.first(:email=>s_addr)
      from_ea    = EmailAlias.first(:email=>f_addr)
      member = if sender_ea && from_ea && sender_ea.owner.eql?(from_ea.owner)
                          raise EmailAlias::ERRORS::Pending if sender_ea.status == 'pending' && from_ea.status == 'pending' && letter[:subject] !~ EmailAlias::CONFIRMING_EMAIL
                          sender_ea.owner
                      elsif sender_ea && from_ea && !sender_ea.owner.eql?(from_ea.owner)
                          raise EmailAlias::ERRORS::OwnerConflict, "#{s_addr} and #{f_addr} are both aliases that belong to two different owners."
                      elsif sender_ea && !from_ea
                        new_alias = f_addr
                        sender_ea.owner
                      elsif !sender_ea && from_ea
                        new_alias = s_addr
                        from_ea.owner
                      else
                        nil
                      end
    end                    
                    
        
    begin
      EmailAlias.__create_and_tell_them__( member, new_alias )
      raise EmailAlias::ERRORS::Pending
    rescue EmailAlias::ERRORS::Approved
      new_alias = nil # Alias is approved and no longer needed.
    end if member && new_alias
    
    raise ERRORS::NoRecordFound, "Sender: #{s_addr}, From: #{f_addr}" unless member
    
    # Validate mail box address ownership. ----------------------------------------
    mb_addr = letter[:to]
    member.mail_box_owner_or_not!( mb_addr )
    letter[:owner_mail_box] = mb_addr
    # ------------------------------------------------------------------------------------------
    
    # Finally, return attach member as owner if everything checks out.
    letter[:owner] = member
    
  end
