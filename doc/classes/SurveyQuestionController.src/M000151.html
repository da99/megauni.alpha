  def update(raw_survey_id, question_id)
    error_msg!("Question not found. Maybe you already deleted it.") unless @question
    begin
      @question.new_values_from_member(clean_room)
      @question.save
      success_msg!(@question.success_msg)
    rescue SurveyQuestion::Invalid
      error_msg!( @question.error_msg )
    end
  end
