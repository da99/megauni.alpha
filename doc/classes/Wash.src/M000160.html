  def self.uri_escape(raw_str)
    require 'URI'
    raw_unsafe_chars         = '\`?!@\#$%^&*()-+=:"|\''
    unsafe_uri_chars         = Regexp.union( URI::UNSAFE, Regexp.new("[#{Regexp.quote(raw_unsafe_chars)}]", false, 'N') )  
    URI::escape(raw_str, unsafe_uri_chars)
  end
