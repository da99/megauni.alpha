  def update(id=0) # PUT
    begin
      @bundle = current_member.bundles_dataset[:id=>clean_room[:id]]
      @bundle.new_values_from_member( clean_room )
      @bundle.save
      success_msg!(@bundle.success_msg, :template=>'__ajax_bundle''__ajax_bundle')
    rescue Bundle::Invalid
      error_msg!(@bundle.error_msg)
    end
  end
