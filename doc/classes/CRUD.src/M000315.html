  def update_it!
    if new?
        raise "RECORD IS NEW: Can not use :update_it!" 
    end
    
    if self.class.columns.include?(:modified_at)
        self.modified_at = Time.now.utc 
    end

    validate_it!

    hook_it(:save){ 
        hook_it(:update) { 
          save
        }
    }
    "#{self.class.english_name.capitalize} has been updated."
  end
