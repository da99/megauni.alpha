  def self.extract_emails( raw_str )
    body = plaintext(raw_str)
    
    body =  if body =~ PARSE::END_EXTRACTION
                    $1
                else
                    body
                end
                
    body.split.inject([[], []]) { |m, txt|
                                                                                  begin
                                                                                    m.first << validate_email_format(txt)
                                                                                  rescue ERRORS::InvalidEmailFormat
                                                                                    m.last << txt if txt['@']
                                                                                  end
                                                                                  m
                                                                            }
  end
