  def find_validation_errors
    # === username of teammate
    if new?
        new_mem = Member[:username=>self.username]
        if new_mem
            self[:member_id] = new_mem[:id]
        else
            errors.add(:member_id, "Member not found. Check spelling.")
        end
    end
    
    # === Title and description
    require_at_least_one_string!(
      :title, :description,
      "Please provide either a title or short description of what this person does."
    )
  end
