  def self.name_underscored(opts = [] )

    valid_opts = [:include_parent_model]
    raise ArgumentError, "Invalid options." unless (opts - valid_opts).empty?
    
    @name_underscored ||= begin
      if self.oo_get(:parent_model) && !opts.include?(:include_parent_model)
        prefix = Kernel.const_get( self.oo_get(:parent_model) ).name.underscore + '_'
        self.name.underscore.sub( prefix, '' )
      else
        self.name.underscore
      end
    end
  end
