  def css_classes(assoc = nil)
    
    no_bundle = self.bundle_id < 1
    no_items  = self.total_items < 1
    no_clones = self.total_clones < 1
    
    arr = []
    
    case assoc
      when nil, :items
        arr << "done" if done?
        arr << "empty" if no_items
        arr << "unbundled" if no_bundle      
      when :clones
        arr << "uncloned" if no_clones
      else
        self.send(assoc)
    end
    
    arr.map { |css| "#{css}_#{self.class.name.underscore}" }.join(' ')
  end
