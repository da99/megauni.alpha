  def success_msg
    return "#{self.class.english_name.capitalize} has not been saved yet." if new?
    return "#{self.class.english_name.capitalize} has not been altered yet." if !@last_known_action

    action = case @last_known_action
      when :create
        "saved"
      when :update
        "saved"
      when :destroy
        "deleted"
      when :trash
        'deleted'
      when :untrash
        'undeleted'
      else
        raise ArgumentError, "Unknown action: #{@last_known_action}"
    end
    
    name = if self.columns.include?(:title)
      ", \"#{self[:title]}\,"
    else
      ""
    end
    
    self.success_msg = "%s %s has been %s." % [self.class.english_name.capitalize, name, action]
    
  end
