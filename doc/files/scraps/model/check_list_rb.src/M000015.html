  def find_validation_errors
    # === title
      if self[:title].blank? && self[:body].blank?
        self.errors.add(:title, "Either a title or body is required.")
      elsif self[:title].blank? && !self[:body].blank?
        self[:title] = self.body.strip.slice(0, 30) + '...' 
      end
    
    # === template_id
      # make sure template belongs to proper owner
      if self.is_clone?
        unless self.owner.eql?( template_dataset.first.owner )
          raise "Mismatch with Template :owner_id --> self.owner_id: #{self.owner_id}, template.owner_id: #{template_dataset.first.owner_id}"
        end
      end
    
    # === bundle_id
      # make sure bundle belongs to proper owner.
      if self.bundle_dataset.first
        unless owner.eql?( bundle_dataset.first.owner )
          raise "Mismatch with Bundle :owner_id --> self.owner_id: #{self.owner_id}, bundle.owner_id: #{bundle_dataset.first.owner_id}"
        end
      end
  end
