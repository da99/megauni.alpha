  def self.validate_email_format(   email_address    ) 
    
    raise( Invalid, "Wrong Class: Email has to be an instance of a String." ) unless email_address.instance_of?(String)
    email_address = email_address.strip.downcase
    
    raise( Invalid, ("Invalid Characters: Email has invalid characters.")) unless email_address =~ VALID_EMAIL_CHARS
    
    raise( Invalid, ('Invalid Format: Email format unrecoginized.')) unless email_address =~ VALID_EMAIL_FORMAT
    
    begin
      validated_email = TMail::Address.parse( email_address ).to_s
    rescue TMail::SyntaxError
      raise( Invalid,  "Invalid Format: Email format could not be recognized."  )
    end
    
    validated_email
  end
